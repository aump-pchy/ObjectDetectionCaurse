import cv2
import matplotlib.pyplot as plt
from google.colab import files

# อัปโหลดภาพจากเครื่อง
uploaded = files.upload()
filename = list(uploaded.keys())[0]

# โหลดภาพด้วย OpenCV (BGR by default)
img_bgr = cv2.imread(filename)
img_rgb = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2RGB)
img_gray = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY)

# แสดงผล 3 รูปเปรียบเทียบ
fig, axes = plt.subplots(1, 3, figsize=(15, 5))
axes[0].imshow(img_rgb);   axes[0].set_title('Original (RGB)')
axes[1].imshow(img_gray, cmap='gray'); axes[1].set_title('Grayscale')
axes[2].imshow(img_rgb[:200,:200]); axes[2].set_title('Cropped (200×200)')

print(f"ขนาดภาพ: {img_rgb.shape}")  # (H, W, C)
plt.tight_layout(); plt.show()
python · Lab 4 — YOLOv8 ครั้งแรก 
!pip install ultralytics -q

from ultralytics import YOLO
from PIL import Image
import matplotlib.pyplot as plt

# โหลด pre-trained model (โหลดอัตโนมัติครั้งแรก ~6MB)
model = YOLO("yolov8n.pt")  # n = nano (เล็กสุด เร็วสุด)

# Detect วัตถุในภาพของเรา
results = model(filename, conf=0.35)

# แสดงผล + พิมพ์สิ่งที่ detect ได้
result = results[0]
result_img = result.plot()  # วาด bounding box บนภาพ

plt.figure(figsize=(12, 8))
plt.imshow(result_img[..., ::-1])
plt.axis('off'); plt.title('YOLOv8 Detection Result')
plt.show()

# รายการสิ่งที่ detect ได้
for box in result.boxes:
    cls = model.names[int(box.cls)]
    conf = float(box.conf)
    print(f"  ✓ {cls:20s} — confidence: {conf:.2%}")
ทดสอบความเข้าใจ
คำศัพท์สำคัญสัปดาห์นี้
📦
Bounding Box
กรอบสี่เหลี่ยมที่ล้อมรอบวัตถุ ระบุด้วยพิกัด (x, y, w, h) หรือ (x1, y1, x2, y2)
📊
Confidence Score
ค่า 0–1 บอกว่าโมเดลมั่นใจแค่ไหนว่า detect ถูกต้อง ค่ายิ่งสูงยิ่งมั่นใจ
🏷️
Class Label
ชื่อของวัตถุที่ detect ได้ เช่น "person", "car", "cat" โมเดล COCO รู้จัก 80 class
🔁
NMS (Non-Max Suppression)
กระบวนการคัด bounding box ที่ซ้อนกันออก เหลือไว้แค่อันที่ดีที่สุดต่อวัตถุ 1 ชิ้น
🧠
Pre-trained Model
โมเดลที่ train บน dataset ใหญ่มาแล้ว (เช่น COCO) ใช้งานได้เลยโดยไม่ต้อง train เอง
⚡
Inference
การนำโมเดลที่ train แล้วไป predict ผลลัพธ์จากข้อมูลใหม่ แตกต่างจากขั้นตอน training
